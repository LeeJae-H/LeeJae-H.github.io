# 단위 테스트 예시
## 0. 테스트에 사용될 객체들
```java
@Entity
@NoArgsConstructor(access = AccessLevel.PROTECTED)
@Getter
public class User {
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @Column(name = "email")
    private String email;

    @Column(name = "password")
    private String password;

    @Column(name = "age")
    private int age;

    @Builder
    public User(String email, String password, int age) {
        this.email = email;
        this.password = password;
        this.age = age;
    }

    public void changeAge(int age) {
        this.age = age;
    }
}
```
```java
public interface UserRepository extends JpaRepository<User,Long> {
    Optional<User> findByEmail(String email);
}
```
```java
@Service
@Transactional
@RequiredArgsConstructor
public class UserService {
    private final UserRepository userRepository;
    
    public Long createUser(String email, String password, int age) {
        userRepository.findByEmail(email)
            .ifPresent(user -> throw new ExpectedException(ErrorCode.ALREADY_EXISTED_USER););

        User user = User.builder()
                        .email(email)
                        .password(password)
                        .age(age)
                        .build();
        return userRepository.save(user).getId();
    }
}
```
```java
@Getter
@RequiredArgsConstructor
public class ExpectedException extends RuntimeException {
    private final ErrorCode errorCode;
}
```
```java
@Getter
@RequiredArgsConstructor
public enum ErrorCode {
    ALREADY_EXISTED_USER("ALREADY_EXISTED_USER", "이미 존재하는 유저입니다.");
    
    private final String errorCode;
    private final String errorMessage;
}
```

## 1. Domain 테스트
```java
public class UserTest {

    @Test
    @DisplayName("유저 생성")
    void createUser() {
        // given
        User user = User.builder().name("joyuri").age(20).build();
    
        // when & then
        assertThat(user.getName()).isEqualTo("joyuri");
        assertThat(user.getAge()).isEqualTo(20);
    }

    @Test
    @DisplayName("유저의 나이 변경")
    void changeUserAge() {
        // given
        User user = User.builder().name("joyuri").age(20).build();
    
        // when
        user.changeAge(25);
        
        // then
        assertThat(user.getAge()).isEqualTo(25);
    }
}
```

## 2. Service 테스트
```java
@ExtendWith(MockitoExtension.class)
public class UserServiceTest {

    @Mock
    private UserRepository userRepository;

    @InjectMocks
    private UserService userService;

    @Test
    @DisplayName("유저 생성 성공")
    void createUser_Success() {
        // given
        User user = User.builder().email("joyuri@gmail.com").password("yuri123").age(20).build();
        ReflectionTestUtils.setField(user, "id" ,3L);
    
        BDDMockito.given(userRepository.save(user))
            .willReturn(user); // Stub

        // when
        Long userId = userService.createUser("joyuri@gmail.com", "yuri123", 20);

        // then
        assertThat(userId).isEqualTo(3L);
    }

    @Test
    @DisplayName("유저 생성 시 이미 존재하는 유저이면 예외 발생")
    void createUser_AlreadyExistedUser_Then_Exception() {
        // given
        User existingUser = User.builder().email("joyuri@gmail.com").password("yuri123").age(20).build();

        BDDMockito.given(userRepository.findByEmail("joyuri@gmail.com"))
            .willReturn(Optional.of(existingUser)); // Stub

        // when & then
        assertThatThrownBy(() -> userService.createUser("joyuri@gmail.com", "yuri123", 20))
            .isInstanceOf(ExpectedException.class)
            .satisfies(e -> {
                assertThat(e.getErrorCode()).isEqualTo(ErrorCode.ALREADY_EXISTED_USER);
            }
        );
    }
}
```
